def compute_com(timesteps):
    com_results = {}
    for timestep, positions in timesteps.items():
        com_x = np.mean(positions[:, 0])
        com_y = np.mean(positions[:, 1])
        com_results[timestep] = (com_x, com_y)
    return com_results

def convert_to_polar(timesteps, com_results):
    polar_results = {}
    for timestep, positions in timesteps.items():
        com_x, com_y = com_results[timestep]
        polar_coords = []
        for x, y in positions:
            r = np.sqrt((x - com_x) ** 2 + (y - com_y) ** 2)
            theta = np.arctan2(y - com_y, x - com_x) #* 180 / np.pi
            if theta < 0:
                theta = 2 * np.pi + theta 
            polar_coords.append((r, theta))
        polar_results[timestep] = np.array(polar_coords)
    return polar_results


def compute_angular_velocity(polar_results, timesteps):
    angular_velocities = {}
    sorted_timesteps = sorted(timesteps.keys())
    
    for i in range(len(sorted_timesteps) - 1):
        t1, t2 = sorted_timesteps[i], sorted_timesteps[i + 1]
        delta_t = (t2 - t1) * 10
        
        theta_t1 = polar_results[t1][:, 1]
        theta_t2 = polar_results[t2][:, 1]
        
        delta_theta = np.unwrap(theta_t2 - theta_t1)
        omega = delta_theta / delta_t
        
        mean_omega = np.mean(omega)
        angular_velocities[t2] = (omega, mean_omega)
    
    return angular_velocities
